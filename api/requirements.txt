# Requirements for running the FastAPI model API (api/main.py).
# This is intentionally separate from the root requirements.txt so the API
# can be deployed (e.g., in a small Docker container) without pulling in
# Jupyter, matplotlib, xgboost, lightgbm, etc.
#
# scikit-learn is PINNED to the exact version used to train and save the
# models (see ../requirements.txt) because joblib.load() needs to
# reconstruct the saved Pipeline/StandardScaler/model objects, and pickled
# objects are sensitive to the library version that created them. If the
# committed .joblib artifacts were trained with xgboost/lightgbm models,
# those packages are also needed here at the same pinned versions for
# joblib.load() to succeed -- uncomment as needed depending on which model
# "won" (check model_used in the API response, or model_name in the
# artifact).

uvicorn[standard]>=0.29.0
pydantic>=2.5.0
joblib>=1.3.0
pandas>=1.5.0
numpy>=1.23.0
scikit-learn==1.9.0
